我们用到的MIPS 3000，虚拟地址的大小是4GB，这4GB应该怎么使用呢？课程组给出的内存图如下：

lab1完成以后，我们已经跳转到了mips_init函数，那么现在，就要对物理内存的初始化做准备了。

内容

2.1的工作简单来说就是以下几个步骤：

1.探测一下我们现在有多少物理内存的空间，MIPS 3000上说有512M可用物理内存，但是可能某种原因有些内存不能用了，所以需要探测一下

2.我们采用了页式管理机制，有不懂的请自己补理论但是简单来说，就是需要把可用的物理内存分成一页一页的，每一页的大小都是4KB，也就是4096字节

3.这些一页一页的物理地址，在虚拟内存里也对应着一个一个page结构体指针，我们需要完成对这些结构体指针的管理

你可能会想，怎么把指针映射到对应的物理内存上？这里课程组已经给了我们相关的代码，我整理了一下，可以方便大家后续的理解阅读：

常用的定义整理

PPN

#define PPN(va) (((u_long)(va)) >> 12)

va这个物理地址对应的页数

PADDR

注意一下宏定义的写法

#define PADDR(kva)                                                                                 \
    ({                                                                                         \
        u_long a = (u_long)(kva);                                                          \
        if (a < ULIM)                                                                      \
            panic("PADDR called with invalid kva %08lx", a);                           \
        a - ULIM;                                                                          \
    })

kva这个虚拟地址对应的物理地址（只在kseg0里）

KADDR

#define KADDR(pa)                                                                                  \
    ({                                                                                         \
        u_long ppn = PPN(pa);                                                              \
        if (ppn >= npage) {                                                                \
            panic("KADDR called with invalid pa %08lx", (u_long)pa);                   \
        }                                                                                  \
        (pa) + ULIM;                                                                       \
    })

pa这个物理地址对应的虚拟地址（只在kseg0里）

pa2page

static inline struct Page *pa2page(u_long pa) {
    if (PPN(pa) >= npage) {
        panic("pa2page called with invalid pa: %x", pa);
    }
    return &pages[PPN(pa)];
}

pa这个物理地址对应的page结构体指针

page2pa

static inline u_long page2pa(struct Page *pp) {
    return page2ppn(pp) << PGSHIFT;
}

pp这个页指针对应的物理地址

page2ppn

static inline u_long page2ppn(struct Page *pp) {
    return pp - pages;
}

pp这个页指针是第几页

page2kva

static inline u_long page2kva(struct Page *pp) {
    return KADDR(page2pa(pp));
}

pp这个页指针对应的虚拟地址

ROUND

#define ROUND(a, n) (((((u_long)(a)) + (n)-1)) & ~((n)-1))

a按照n向上对齐，举个例子就是如果a是102，n是4，那么返回的结果应该是104，也就是说102~104这部分被弃用了x

BY2PG

4096……就是一个物理页的大小

个人感觉2.1还是分为两部分的

第一部分是分配一下存储指针的虚拟空间

第二部分是补一下这些虚拟指针的相关代码

2.1.1

mips_detect_memory()

 void mips_detect_memory() {
         /* Step 1: Initialize memsize. */
         memsize = *(volatile u_int *)(KSEG1 | DEV_MP_ADDRESS | DEV_MP_MEMORY);
 
        /* Step 2: Calculate the corresponding 'npage' value. */
         /* Exercise 2.1: Your code here. */
        npage = memsize >> PGSHIFT; //这是memsize/4096的意思 PGSHIFT是12
        printk("Memory size: %lu KiB, number of pages: %lu\n", memsize / 1024, npage);
 }

很简单，就是探测可用内存，单位是字节，然后/4096（一个物理页的大小）

mips_vm_init()

void mips_vm_init() {
       /* Allocate proper size of physical memory for global array `pages`,
          * for physical memory management. Then, map virtual address `UPAGES` to
          * physical address `pages` allocated before. For consideration of alignment,
         * you should round up the memory size before map. */
         pages = (struct Page *)alloc(npage * sizeof(struct Page), BY2PG, 1); //BY2PG就是4096
        printk("to memory %x for struct Pages.\n", freemem);
         printk("pmap.c:\t mips vm init success\n");
 }

我们只从含义上分析

上文中计算出了物理页数npage，我们从kseg0中的某个地址开始，分配了npage个struct page结构体的大小

每个结构体对应了一个物理页，从而实现对虚拟内存对物理内存的管理

下面再看看alloc函数，长这样

简单概括一下用法就是，n是要分配的虚拟空间的大小，align是对齐单位，clear是是否置零

分配一段虚拟地址，用于存储管理物理页面的指针

void *alloc(u_int n, u_int align, int clear) {
    extern char end[];
    u_long alloced_mem;

    /* Initialize `freemem` if this is the first time. The first virtual address that the
     * linker did *not* assign to any kernel code or global variables. */
    if (freemem == 0) {
        freemem = (u_long)end; // end
    } //初始化的意思，就是说我们分配的空间应该从0x80400000

    /* Step 1: Round up `freemem` up to be aligned properly */
    freemem = ROUND(freemem, align); //根据上面的宏定义，对齐freemem 

    /* Step 2: Save current value of `freemem` as allocated chunk. */
    alloced_mem = freemem; //设置分配的初始地址

    /* Step 3: Increase `freemem` to record allocation. */
    freemem = freemem + n; //找到分配的终止地址

    // Panic if we're out of memory.
    panic_on(PADDR(freemem) >= memsize); //如果这个虚拟地址对应的物理地址超过了我们上面探测出的可用的硬件地址，就会返回报错

    /* Step 4: Clear allocated chunk if parameter `clear` is set. */
    if (clear) {
        memset((void *)alloced_mem, 0, n);
    } //如果clear,就把分配的这段虚拟地址全部变成0

    /* Step 5: return allocated chunk. */
    return (void *)alloced_mem;
}

2.1.3

下面就要正式进行物理页面的管理了，也就是把虚拟页面和物理页面一一对应起来

那么就来逐个看看课程组给出的函数吧

page_init

 void page_init(void) {
         LIST_INIT(&page_free_list); //就是上文的page_list结构体
         ROUND(freemem,BY2PG); //上面的mips_vm_init分配了n空间的内存，这里要把freemem对齐，freemem以前的空间都已经被分配
         struct Page *now;
         for(now = pages;page2kva(now) < freemem;now++){
                 now -> pp_ref = 1;
         } //上面的每个虚拟页面的引用次数都变成1
         for(now = &pages[PPN(PADDR(freemem))];page2ppn(now) < npage;now++) {
                 now -> pp_ref = 0;
                 LIST_INSERT_HEAD(&page_free_list,now,pp_link);
         } //freemem以下的页面，插入到空闲页面里面
 }

page_alloc

通过page_init，我们的所有物理页面已经被分成了两类，一类是使用过的，一类是没有被使用的,没有被使用过的被放在了空闲链表。每当page_alloc的时候，我们都会从空闲链表里取出一页

 int page_alloc(struct Page **new) {
         struct Page *pp;
         if (LIST_EMPTY(&page_free_list)) {
                 return -E_NO_MEM;
         }
         pp = LIST_FIRST(&page_free_list);
         LIST_REMOVE(pp, pp_link);
         memset(page2kva(pp),0,4096);//这一步之前错了，把pp对应的物理页面置0
         *new = pp;
         return 0;
 }

page_decref

 void page_decref(struct Page *pp) {
         assert(pp->pp_ref > 0); //这个是如果不符号条件会报错
         /* If 'pp_ref' reaches to 0, free this page. */
         if (--pp->pp_ref == 0) {
                 page_free(pp);
         }
 }

page_free

就是把这个页面重新插空闲页面

 void page_free(struct Page *pp) {
         assert(pp->pp_ref == 0); //如果为假会报错
         LIST_INSERT_HEAD(&page_free_list,pp,pp_link);
 }
